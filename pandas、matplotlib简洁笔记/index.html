    Series:一维

    import pandas as pd
from pandas import Series,DataFrame

# Series,创建可以传入:list,dict,默认索引从0~N-1，可用index指定索引名
obj = pd.Series([4,7,-5,3]) # 普通创建:(索引,值)集合
obj = pd.Series([4,7,-5,3],index=['d','b','a','c']) # 指定索引
obj1 = pd.Series({'a':1,'b':2,'c':3}) # index:a,b,c values:1,2,3
obj1 = pd.Series({'a':1,'b':2,'c':3},index=['c','b','a']) # 使用index指定顺序
obj2 = obj1.reindex(['b','c','a']) # 重建索引
# 重建索引高级用法
obj3 = pd.Series(['blue','purple','yellow'],index=[0,2,4])
obj3.reindex(range(6),method='ffill')# ffill是重建索引时向前插值，复制前方值当做己方值

obj + obj1 # Series的数学操作会根据索引自动对齐，有共同索引的执行数学运算，无的value=NaN

# 属性+方法
obj.index # get/set所有索引
obj.values # 获取所有值
obj.index.name # get/set索引名,因为values返回的是array对象，所以没有.name属性
obj.index.is_unique # 判断索引名是否是唯一的，含有重复吗
obj.value_counts() # 常用:返回每个value出现的次数(value:times)
pd.value_counts(obj.values,sort=False)# 更加常用
uniques = obj.unique() # 常用:类似set()

obj['d']==obj[0] # 获取单个值，索引是‘d’的值
obj[['b','a','c']] # 获取部分值

obj[2:4] # 虽然索引改成了a,b,c,d,但是照样可以使用整数索引获取，所以可以直接使用切片
obj['b':'c'] # 当使用索引切片时，注意:这个切片包含着尾部，和数组切片不同
# 整数索引和名称索引傻傻分不清，以后规定，想要切片统一使用iloc函数，这样就能像numpy一样骚操作了
obj.iloc[:] # 等价于整数索引，不过以后取值统一加上iloc

'b' in obj # True
mask = obj.isin(['b','c'])# 返回一个mask模板，obj[mask] #这样就可以选择出来了
new_obj = obj.drop('a') # 删除索引a这一值
new_obj = obj.drop(['a','b']) # 删除多值

obj[obj>0] # 通过值进行筛选
obj*2 # 对值进行增倍
np.exp(obj) # 可以对Series进行操作

# 判断缺失数据,根据value值进行判断
pd.isnull(obj) == obj.isnull() # return (index,bools)
pd.notnull(obj)


# 排序
obj.sort_index() # 按照索引排序
obj.sort_values() # 按值进行排序，所有缺失值都会被排序到Series的尾部


# 其他常用方法
obj.str.lower()
obj.replace([4,7],np.nan)
obj.replace([4,7],[999,1000])
obj.replace({4:999,7:1000})
obj.rename(index={'a':'aa'},inplace=True)

    DataFrame:二维

    import pandas as pd
from pandas import Series,DataFrame

# DataFrame,创建可以传入:dict,默认索引从0~N-1,可用index指定索引名,columns指定列名
# dict{key:list}创建
frame = pd.DataFrame({'colN1':[1,2,3,4], # 列名:values
                      'colN2':[1,2,3,4],
                      'colN3':[1,2,3,4],},
                      index=['index1','index2','index3','index4'], # 指定索引名
                      columns=['colN2','colN3','colN1']) # 指定列名排列顺序
# dict{key:dict} # 创建
frame1 = pd.DataFrame({'colN1':{'index1':1,'index2':1},
                       'colN2':{'index1':2,'index2':2}})
# dict{key:Series}
frame2 = pd.DataFrame({'colN1':frame['colN1'],
                       'colN2':frame['colN2']})
# 重建索引
frame.reindex(index=[...],columns=[...])
# 更多人用下面方式更改标签索引
frame.loc[[indexs],[columns]]


# 属性
frame.head() # 前五行
frame.index # get所有索引，注意:索引对象是无法修改的，这样保证了安全性。
frame.index.name # get/set索引名
frame.columns # get/set所有列
frame.columns.name # get/set列名
frame.values # 获取所有值(return:二维array)
frame.T # 行列转换


# 获取和操作数值
frame['colN1'] == frame.colN1 # 选择某列,注意是列名

frame.loc['index1'] # 选择某行，行的选择比较复杂
frame.loc[['index1','index2']] # 选择多行，还有更简单的方法，切片
frame[:2] # 行选择,切片使用，和numpy数组类似，这个指前两行
# 强大的数据选择方法iloc,使用iloc后就能像numpy一样为所欲为
frame.iloc[2] # [index]
frame.iloc[2,[3,0,1]] #[index,[columns]]
frame.iloc[[1,2],[3,0,1]] # [[index],[columns]]
frame.iloc[:,:3][frame['colN1']>1]

frame[frame['colN1']>1] # 按值筛选
frame['colN1'][:2] # 选择这列前两行元素 
frame['colNull'] = 5 # 新建某列，并赋值，这里可以广播的
frame['colNull'] = np.arange(4) # 修改某列值
frame['colNull'] = pd.Series([5,5,5,5],index=['index1','index2','index3','index4'])
del frame['colNull'] # 删除某列
new_frame = frame.drop(['index1','index2']) # 删除两行,默认axis=0,箭头由上往下，指行:按行处理
new_fame = frame.drop(['colN2'],axis=1) # 删除某列，axis=1,箭头由左往右，指列:按列处理
frame.drop(...,inplace=True) # 推荐使用，直接在原对象中修改


# 详解索引对象,注意:索引对象里面的值是可以重复的
index1 = pd.Index(['foo','foo','bar'])
index2 = index1.append(pd.Index(['bar','test'])) # 添加额外索引到原索引，返回新索引
index2.difference(index1) # 差集
index2.intersection(index1) # 交集
index2.union(index1) # 并集
pd.Index(['test']).isin(index1)
index1.delete(index) # 返回删除的值，注意原先的索引没有删除


# 填充
frame1+frame2 # 索引和列名会自动对应，没有对应上的会用NaN进行填充
# 解决方案
frame1.add(frame2,fill_value=0) # 加法
frame1.sub(frame2,fill_value=0) # 减法
frame1.div(frame2,fill_value=0) # 除法
frame1.floordiv(frame2,fill_value=0) # 整除
frame1.mul(frame2,fill_value=0) # 乘法
frame1.pow(frame2,fill_value=0) # 幂次方


# apply函数:把某一函数应用到一行或一列的一维数组上
frame.apply(lambda x:x.max() - x.min()) # 默认axis=0,按照行计算
frame.apply(lambda x:x.max() - x.min(),axis=1) # 按照列计算
# applymap函数:是逐元素计算的。注意:Series里面也有类似的函数:map()
frame.applymap(lambda x:f'{x:.2f}') # 所有元素保留两位小数

# 排序
frame.sort_index() # 默认axi=0
frame.sort_index(axis=1)
frame.sort_index(axis=1,ascending=False) # 降序
frame.sort_values(by=['col1','col2']) # 按照某列/多列进行排序


# 统计:NaN默认是被排除的,可以使用skipna=False,来保留NaN
frame.sum()/frame.sum(axis=xx) 
frame.mean()/frame.mean(axis=xx,)
frame.idxmax(axis=xx)/frame.idxmin(axis=xx) # 统计最小值或最大值的索引值
frame.cumsum() # 累加型
frame.describe() # 汇总统计，常用


# 缺失值删除
obj.dropna() == obj[obj.notnull()] # 直接删除值是na的行
frame.dropna(axis=xx) # 严格:含有NaN的行就删除,axis=1,就是删除列
frame.dropna(how='all') # 宽松:这行所有值均为NaN才删除
frame.dropna(thresh=2)# 筛选一下:删除>=thresh个NaN的行
# 补全缺失值
frame.fillna(0,inplace=True) # 全部补全为0,inplace操作
frame.fillna({"colN1":0,"colN2":1}) # 为不同列补全为不同的值
frame.fillna(method='ffill') # 用上面的值代替自己的NaN
frame.fillna(method='ffill',limit=2)# 用前向或后向填充时最大的填充范围
frame.fillna(frame.mean()) # 使用Series的平均值进行补充
# 删除重复值:默认保留的第一个值，如果想保留最后一个值:keep='last'
frame.duplicated() # return:每行bool,反应是否含有重复行
frame.drop_duplicates() # 删除重复行，默认是:两行所有列的值都重复才算重复
frame.drop_duplicates(['colN1'])# 只要colN1列的值重复就删除


# 其他重用方法
# 重命名方法
frame.rename(index={'index1':'new_index1'},columns={'colN1':'new_colN1'},inplace=True)
# 随机排列frame的行顺序
frame.take(np.random.permutation(frame.shape[0]))

axes[0,0]# 选取第一个子图

# 子图的方法
plt.subplots(
    nrows=1, # 子图行数
    ncols=1,# 子图列数
    sharex=False,# 所有子图使用相同的x轴刻度
    sharey=False,# 所有子图使用相同y轴刻度
    squeeze=True,
    subplot_kw=None,
    gridspec_kw=None,
    **fig_kw, # 额外参数:figsize=(10,8)
)
# 调整子图周围间距

    bottom=None,
    right=None,

import numpy as np
import matplotlib.pyplot as plt

# 创建画笔和子图
fig,axes = plt.subplots(2,2,sharex=True,sharey=True,figsize=(8,6))
axes[0,0].plot(np.arange(0,5),np.arange(0,5),'ko--',label='line') 
axes[0,0].plot(np.random.randn(80),'g--',drawstyle='steps-post',label='step')

axes[0,0].set_xticks([0,40,80]) #表示在数据范围内设定刻度的位置
# 为x刻度设置label
axes[0,0].set_xticklabels(['one','two','three'],rotation=30,fontsize='small')

axes[0,0].set_title('line') # 设置名称
axes[0,0].set_xlabel("x-line") # 设置x轴名称
# 等价于
props = {"title":'line','xlabel':'x-line'}
axes[0,0].set(**props)

# 添加注释和文本:坐标(x,y),test/arrow/annote
axes[0,0].text(4,3,'This is a line') # 文本注释

axes[0,0].legend(loc='best') # 显示label
# plt.subplots_adjust(wspace=0,hspace=0) # 调整子图之间的间距
